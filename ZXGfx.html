    class ZXGfx

    A module with Z80 Macros for common ZX Spectrum graphics tasks¶ ↑

    Example:

    require 'zxlib/gfx'

class Program
  include Z80
  macro_import  ZXGfx

          # invert pixel at 100, 100
          ld  l, 100
          ld  a, 100
          xytoscr(a, l, h, l, b, c)
          inc b
          ld  a, 1
  shift1  rrca
          djnz shift1
          xor  [hl]
          ld   [hl], a
          ret
end
