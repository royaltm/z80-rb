def instrument(name, &block)
  first_octave_note_ = @first_octave_note
  tempo_ = @tempo
  @instruments[name.to_sym] = Class.new do |klass|
    include Instrument
    tempo(tempo_)
    klass.module_eval(&block)
  end
end
